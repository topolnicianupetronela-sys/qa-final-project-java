CLASA ApiTests 
 
    // Variabilă definită la nivel de clasă pentru a stoca URL-ul de bază 
al serviciului. 
    // Va fi accesibilă în toate testele din această clasă. 
    VARIABILA baseUrl DE TIP Text 
 
    // Procedură specială care se execută o singură dată, înainte de a rula 
orice test din clasă. 
    // Echivalentul lui @BeforeClass. 
    PROCEDURA setup() 
        // Încearcă să încarci și să citești fișierul de configurare 
"app.yaml". 
        Încearcă 
            dateConfigurare = CiteșteFișierYAML("config/app.yaml") 
        // Dacă fișierul nu poate fi găsit sau citit, oprește execuția și 
afișează o eroare. 
        CATCH EroareLaCitireFișier 
            ARUNCĂ EROARE "Nu s-a putut găsi sau citi fișierul de 
configurare 'app.yaml'" 
        SFÂRȘIT Încearcă 
 
        // Extrage valoarea specifică din datele de configurare citite. 
        // Caută obiectul 'service', apoi în interiorul lui caută cheia 
'baseUrl'. 
        baseUrl = ExtrageValoare(dateConfigurare, "service.baseUrl") 
    SFÂRȘIT PROCEDURA 
 
     // Test principal: verifică status 200 și câmpul 'title'
    TEST test_verifica_todo_dupa_title()
        // == ARRANGE ==
        endpoint = "/todos/1"
        urlComplet = CombinăText(baseUrl, endpoint)
        AfișeazăLaConsolă("Se trimite GET request către: " + urlComplet)

 
         // == ACT ==
        răspunsApi = TrimiteCerereGET(la_adresa: urlComplet)

        // == ASSERT ==
        Verifică(răspunsApi.codDeStare ESTE EGAL CU 200)
        Verifică(răspunsApi.corp.conține_cheia("title"))

        AfișeazăLaConsolă("Testul a trecut cu succes: cod 200 și câmpul 'title' este prezent.")
    SFÂRȘIT TEST
